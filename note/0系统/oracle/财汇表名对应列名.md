# 具体表

```sql
-- 查T_DI_CONFIG的fid
SELECT fid,fcode,fname,FTABLENAME FROM T_DI_CONFIG 
WHERE upper(FDB_SQL) LIKE UPPER('%BondTypes%') 

-- 根据fid查字段
SELECT d.fname 资产托管接口名称,d.FTABLENAME 资产托管表名, f.FFILENAME 财汇列名 ,f.fname 资产托管列名,
(SELECT c.COMMENTS
          FROM ALL_COL_COMMENTS c
         WHERE UPPER(c.TABLE_NAME) = UPPER(d.FTABLENAME)
           AND UPPER(c.COLUMN_NAME) = UPPER(f.fname)
           AND ROWNUM = 1) AS 资产托管列中文名
FROM t_di_config d
INNER JOIN T_DI_CONFIG_FIELD f ON d.fid=f.FCONFIG_ID 
WHERE 
d.fid=18042422031429000968
AND d.FCODE NOT LIKE '%\_LS' ESCAPE '\'
AND d.fdeleted=0
ORDER BY d.FTABLENAME,f.FFILENAME
```



# 总的

```sql
t_temp,fname

SELECT e.fname 财汇表名,d.fname 资产托管接口名称,d.FTABLENAME 资产托管表名, f.FFILENAME 财汇列名 ,f.fname 资产托管列名,
(SELECT c.COMMENTS
          FROM ALL_COL_COMMENTS c
         WHERE UPPER(c.TABLE_NAME) = UPPER(d.FTABLENAME)
           AND UPPER(c.COLUMN_NAME) = UPPER(f.fname)
           AND ROWNUM = 1) AS 资产托管列中文名
FROM t_Temp e
INNER JOIN t_di_config d ON upper(d.FDB_SQL) LIKE upper('%'||e.fname|| '%') AND d.fcode=e.fname
INNER JOIN T_DI_CONFIG_FIELD f ON d.fid=f.FCONFIG_ID 
WHERE d.FCODE NOT LIKE '%\_LS' ESCAPE '\'
AND d.fdeleted=0
ORDER BY e.fname,f.FFILENAME
```
